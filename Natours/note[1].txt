building process
1. compile:sass: node-sass sasss/main.scss css/style.comp.css
2. in terminal 'npm install concat --save-dev' this is the concatnation package
3. after we type "concat:css": "concat -o css/style.concat.css  css/icon-font.css   css/style.comp.css" in package.json
   in terminal 'npm run concat:css'

4. step 4 terminal 'npm install autoprefixer -g' and 'npm install postcss-cli -g'
     then terminal 'npm run prefix:css'
5. type : '"compress:css": "node-sass css/style.prefix.css css/style.css  --output-style compressed"'
         then npm run compress:css

6. last terminal 'npm install npm-run-all --save-dev'
